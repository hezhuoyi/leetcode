## 20. 有效的括号
给定一个只包括 '('，')'，'{'，'}'，'['，']' 的字符串，判断字符串是否有效。
有效字符串需满足：
左括号必须用相同类型的右括号闭合。
左括号必须以正确的顺序闭合。
注意空字符串可被认为是有效字符串。

```js
var isValid = function(s) {
    let i = 0
    let stack = []
    while (i < s.length) {
        let cur = s[i]
        if (cur === '{' || cur === '[' || cur === '(') {
            stack.push(cur)
        } else if (cur === '}' || cur === ']' || cur === ')') {
            if (!stack.length) return false
            let leftEnd = stack.pop()
            if (leftEnd === '{' && cur !== '}') return false
            if (leftEnd === '[' && cur !== ']') return false
            if (leftEnd === '(' && cur !== ')') return false
        }
        i++
    }
    return !stack.length
};
```
