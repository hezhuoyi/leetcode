## 144. 二叉树的前序遍历
给你二叉树的根节点 root ，返回它节点值的 前序 遍历。

```js
// 递归 深度优先搜索
var preorderTraversal = function(root) {
    let res = []
    let preorder = (node) => {
        if (!node) return
        res.push(node.val)
        preorder(node.left)
        preorder(node.right)
    }
    preorder(root)
    return res
};

// 迭代 广度优先搜索
var preorderTraversal = function(root) {
    let res = []
    let stack = []
    if (root) stack.push(root)
    while (stack.length) {
        let head = stack.pop()
        res.push(head.val)
        // 栈是先入后出，前序是 根 左 右，所以先压入右节点，后入左节点
        if (head.right) stack.push(head.right)
        if (head.left) stack.push(head.left)
    }
    return res
};
```
